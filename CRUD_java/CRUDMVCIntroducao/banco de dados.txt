    id      INT         NOT NULL PRIMARY KEY AUTO_INCREMENT,
    Name_    VARCHAR(50) NOT NULL,
    Email   VARCHAR(50) NOT NULL,
    Phone VARCHAR(11) NOT NULL,
    Postal_Code     VARCHAR(9)  NOT NULL
);
